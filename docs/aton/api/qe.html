  47def read_in(filepath) -> dict:
  48    """Reads a Quantum ESPRESSO input `filepath` and returns the values as a dict.
  49
  50    Dict keys are named after the corresponding variable.
  51    CARDS are returned as lists, and contain the
  52    title card + parameters in the first item.
  53    """
  54    file_path = file.get(filepath)
  55    data: dict = {}
  56    # First get the values from the namelists
  57    lines = find.lines(file_path, '=')
  58    for line in lines:
  59        line.strip()
  60        var, value = line.split('=', 1)
  61        var = var.strip()
  62        value = value.strip()
  63        if var.startswith('!'):
  64            continue
  65        try:
  66            value_float = value.replace('d', 'e')
  67            value_float = value_float.replace('D', 'e')
  68            value_float = value_float.replace('E', 'e')
  69            value_float = float(value_float)
  70            value = value_float
  71            if var in _pw_int_values: # Keep ints as int
  72                value = int(value)
  73        except ValueError:
  74            pass # Then it is a string
  75        data[var] = value
  76    # Try to find all the cards. Card titles will be saved in the 0 position of each result.
  77    for card in pw_cards.keys():
  78        card_lower = card.lower()
  79        card_uncommented = rf'(?!\s*!\s*)({card}|{card_lower})'
  80        card_content = find.between(filepath=file_path, key1=card_uncommented, key2=_all_cards_regex, include_keys=True, match=-1, regex=True)
  81        if not card_content:
  82            continue
  83        # If found, clean and normalise the card's content
  84        card_content = card_content.splitlines()
  85        card_content = normalize_card(card_content)
  86        # Ignore OCCUPATIONS card if not required to avoid saving nonsense data
  87        if card_lower == 'occupations':
  88            if data.get('occupations') is None or 'from_input' not in data.get('occupations'):
  89                continue
  90        data[card] = card_content
  91    # If there are CELL_PARAMETERS, check if we can extract the alat to celldm(1).
  92    if 'CELL_PARAMETERS' in data.keys():
  93        if 'alat' in data['CELL_PARAMETERS'][0]:
  94            alat = extract.number(data['CELL_PARAMETERS'][0])
  95            if alat:  # This overwrites any possible celldm(1) previously defined!
  96                data['celldm(1)'] = alat
  97                data['CELL_PARAMETERS'][0] = 'CELL_PARAMETERS alat'
  98    return data

 101def read_out(filepath) -> dict:
 102    """Reads a Quantum ESPRESSO output `filepath`, returns a dict with the output keys.
 103
 104    The output keys are:
 105    `'Energy'` (Ry), `'Total force'` (float), `'Total SCF correction'` (float),
 106    `'Runtime'` (str), `'JOB DONE'` (bool), `'BFGS converged'` (bool), `'BFGS failed'` (bool),
 107    `'Maxiter reached'` (bool), `'Error'` (str), `'Success'` (bool), `'CELL_PARAMETERS out'` (list of str), `'ATOMIC_POSITIONS out'` (list of str), `'Alat'` (bohr), `'Volume'` (a.u.^3), `'Density'` (g/cm^3).
 108
 109    Note that these output keys start with a **C**apital letter.
 110    """
 111    file_path = file.get(filepath)
 112
 113    energy_key           = '!    total energy'
 114    force_key            = 'Total force'
 115    scf_key              = 'Total SCF correction'
 116    time_key             = 'PWSCF'
 117    time_stop_key        = 'CPU'
 118    job_done_key         = 'JOB DONE.'
 119    bfgs_converged_key   = 'bfgs converged'
 120    bfgs_failed_key      = 'bfgs failed'
 121    maxiter_reached_key  = r'(Maximum number of iterations reached|maximum number of steps has been reached)'
 122    error_key            = 'Error in routine'
 123    error_failed_line    = 'pw.x: Failed'
 124    cell_parameters_key  = 'CELL_PARAMETERS'
 125    atomic_positions_key = 'ATOMIC_POSITIONS'
 126
 127    energy_line          = find.lines(file_path, energy_key, -1)
 128    force_line           = find.lines(file_path, force_key, -1)
 129    time_line            = find.lines(file_path, time_key, -1)
 130    job_done_line        = find.lines(file_path, job_done_key, -1)
 131    bfgs_converged_line  = find.lines(file_path, bfgs_converged_key, -1)
 132    bfgs_failed_line     = find.lines(file_path, bfgs_failed_key, -1)
 133    maxiter_reached_line = find.lines(file_path, maxiter_reached_key, -1, regex=True)
 134    error_line           = find.lines(file_path, error_key, -1, 1, True)
 135    error_failed_line    = find.lines(file_path, error_failed_line, -1)
 136
 137    energy: float = None
 138    force: float = None
 139    scf: float = None
 140    time: str = None
 141    job_done: bool = False
 142    bfgs_converged: bool = False
 143    bfgs_failed: bool = False
 144    maxiter_reached: bool = False
 145    error: str = ''
 146    success: bool = False
 147
 148    if energy_line:
 149        energy = extract.number(energy_line[0], energy_key)
 150    if force_line:
 151        force = extract.number(force_line[0], force_key)
 152        scf = extract.number(force_line[0], scf_key)
 153    if time_line:
 154        time = extract.string(time_line[0], time_key, time_stop_key)
 155    if job_done_line:
 156        job_done = True
 157    if bfgs_converged_line:
 158        bfgs_converged = True
 159    if bfgs_failed_line:
 160        bfgs_failed = True
 161    if maxiter_reached_line:
 162        maxiter_reached = True
 163    if error_line:
 164        error = error_line[1].strip()
 165    elif error_failed_line:
 166        error = error_failed_line[0].strip()
 167
 168    # Was the calculation successful?
 169    if job_done and not bfgs_failed and not maxiter_reached and not error:
 170        success = True
 171
 172    # CELL_PARAMETERS and ATOMIC_POSITIONS
 173    alat = None
 174    volume = None
 175    density = None
 176    cell_parameters = None
 177    atomic_positions = None
 178    cell_parameters_raw = []
 179    atomic_positions_raw = []
 180    coordinates_raw = find.between(file_path, 'Begin final coordinates', 'End final coordinates', False, -1, False)
 181    if coordinates_raw:
 182        coordinates_raw = coordinates_raw.splitlines()
 183        append_cell = False
 184        append_positions = False
 185        for line in coordinates_raw:
 186            line = line.strip()
 187            if cell_parameters_key in line:
 188                append_cell = True
 189                append_positions = False
 190            elif atomic_positions_key in line:
 191                append_cell = False
 192                append_positions = True
 193            elif 'volume' in line:
 194                volume = extract.number(line, 'volume')
 195            elif 'density' in line:
 196                density = extract.number(line, 'density')
 197            if line == '' or line.startswith('!'):
 198                continue
 199            if append_cell:
 200                cell_parameters_raw.append(line)
 201            elif append_positions:
 202                atomic_positions_raw.append(line)
 203        atomic_positions = normalize_card(atomic_positions_raw)
 204        if cell_parameters_raw:
 205            cell_parameters = normalize_card(cell_parameters_raw)
 206            if 'alat' in cell_parameters[0]:
 207                alat = extract.number(cell_parameters[0], 'alat')
 208    # If not found, at least try to keep the latest iteration
 209    else:
 210        _nat_raw = find.lines(file_path, 'number of atoms/cell', 1)
 211        if _nat_raw:
 212            _nat = int(extract.number(_nat_raw[0], 'number of atoms/cell'))
 213            atomic_positions_raw = find.lines(file_path, atomic_positions_key, matches=-1, additional=_nat, split=True)
 214        if atomic_positions_raw:
 215            atomic_positions = normalize_card(atomic_positions_raw)
 216        cell_parameters_raw = find.lines(file_path, cell_parameters_key, matches=-1, additional=3, split=True)
 217        if cell_parameters_raw:
 218            cell_parameters = normalize_card(cell_parameters_raw)
 219            if 'alat' in cell_parameters[0]:
 220                alat = extract.number(cell_parameters[0], 'alat')
 221
 222    output = {
 223        'Energy'                : energy,
 224        'Total force'           : force,
 225        'Total SCF correction'  : scf,
 226        'Runtime'               : time,
 227        'JOB DONE'              : job_done,
 228        'BFGS converged'        : bfgs_converged,
 229        'BFGS failed'           : bfgs_failed,
 230        'Maxiter reached'       : maxiter_reached,
 231        'Error'                 : error,
 232        'Success'               : success,
 233        'CELL_PARAMETERS out'   : cell_parameters,
 234        'ATOMIC_POSITIONS out'  : atomic_positions,
 235        'Alat'                  : alat,
 236        'Volume'                : volume,
 237        'Density'               : density,
 238    }
 239    return output

 242def read_dir(
 243        folder=None,
 244        in_str:str='.in',
 245        out_str:str='.out'
 246    ) -> dict:
 247    """Takes a `folder` from a QE calculation, returns a dict with input and output values.
 248    
 249    Input and output files are determined automatically,
 250    but must be specified with `in_str` and `out_str`
 251    if more than one file ends with `.in` or `.out`.
 252    """
 253    folder = file.get_dir(folder)
 254    input_file = file.get(folder, in_str)
 255    output_file = file.get(folder, out_str)
 256    if not input_file and not output_file:
 257        return None
 258    if input_file:
 259        dict_in = read_in(input_file)
 260        if not output_file:
 261            return dict_in
 262    if output_file:
 263        dict_out = read_out(output_file)
 264        if not input_file:
 265            return dict_out
 266    # Merge both dictionaries
 267    merged_dict = {**dict_in, **dict_out}
 268    return merged_dict

 271def read_dirs(
 272        directory,
 273        in_str:str='.in',
 274        out_str:str='.out',
 275        calc_splitter='_',
 276        calc_type_index=0,
 277        calc_id_index=1
 278    ) -> None:
 279    """Reads recursively QE calculations from all the subfolders inside the given `directory`.
 280
 281    Results are saved to CSV files inside the current directory.
 282    Input and output files are determined automatically, but must be specified with
 283    `in_str` and `out_str` if more than one file ends with `.in` or `.out`.
 284
 285    To properly group the calculations per type, saving separated CSVs for each calculation type,
 286    you can modify `calc_splitter` ('_' by default), `calc_type_index` (0) and `calc_id_index` (1).
 287    With these default values, a subfolder named './CalculationType_CalculationID_AdditionalText/'
 288    will be interpreted as follows:
 289    - Calculation type: 'CalculationType' (The output CSV will be named after this)
 290    - CalculationID: 'CalculationID' (Stored in the 'ID' column of the resulting dataframe)
 291
 292    If everything fails, the subfolder name will be used for the CSV file.
 293    """
 294    print(f'Reading all Quantum ESPRESSO calculations from {directory} ...')
 295    folders = file.get_list(directory)
 296    if not folders:
 297        raise FileNotFoundError('The directory is empty!')
 298    # Separate calculations by their title in an array
 299    calc_types = []
 300    folders.sort()
 301    for folder in folders:
 302        if not os.path.isdir(folder):
 303            folders.remove(folder)
 304            continue
 305        folder_name = os.path.basename(folder)
 306        try:
 307            calc_name = folder_name.split(calc_splitter)[calc_type_index]
 308        except:
 309            calc_name = folder_name
 310        if not calc_name in calc_types:
 311            calc_types.append(calc_name)
 312    len_folders = len(folders)
 313    total_success_counter = 0
 314    for calc in calc_types:
 315        len_calcs = 0
 316        success_counter = 0
 317        results = pd.DataFrame()
 318        for folder in folders:
 319            if not calc in folder:
 320                continue
 321            len_calcs += 1
 322            folder_name = os.path.basename(folder)
 323            try:
 324                calc_id = folder_name.split(calc_splitter)[calc_id_index]
 325            except:
 326                calc_id = folder_name
 327            df = pd.DataFrame.from_dict(read_dir(folder, in_str, out_str))
 328            if df is None:
 329                continue
 330            # Join input and output in the same dataframe
 331            df.insert(0, 'ID', calc_id)
 332            df = df.dropna(axis=1, how='all')
 333            results = pd.concat([results, df], axis=0, ignore_index=True)
 334            if df['Success'][0]:
 335                success_counter += 1
 336                total_success_counter += 1
 337        results.to_csv(os.path.join(directory, calc+'.csv'))
 338        print(f'Saved to CSV: {calc} ({success_counter} successful calculations out of {len_calcs})')
 339    print(f'Total successful calculations: {total_success_counter} out of {len_folders}')

 342def set_value(
 343        filepath,
 344        key:str,
 345        value,
 346        indent:str='  ',
 347    ) -> None:
 348    """Replace the `value` of a `key` parameter in an input `filepath`.
 349    
 350    Delete parameters with `value=''`.
 351    Remember to include the single quotes `'` on values that use them.
 352
 353    Updating 'ATOMIC_POSITIONS' updates 'nat' automatically,
 354    and updating 'ATOMIC_SPECIES' updates 'ntyp'.
 355
 356    Optionally change indentation with `indent`, 2 spaces by default.
 357    """
 358    key = key.strip()
 359    file_path = file.get(filepath)
 360    input_old = read_in(file_path)
 361    # Present keys uppercase
 362    present_keys_upper = []
 363    for present_key in input_old.keys():
 364        present_keys_upper.append(present_key.upper())
 365    # All namelist values
 366    pw_values = []
 367    for namelist_values in pw_namelists.values():
 368        pw_values.extend(namelist_values)
 369    # All cards
 370    pw_cards_upper = []  # Don't forget about these!
 371    for card in pw_cards.keys():
 372        pw_cards_upper.append(card.upper())
 373    # Check if it's a namelist
 374    if key in pw_values:
 375        if key in input_old.keys():
 376            _update_value(filepath, key, value, indent)
 377        else:  # Write the value from scratch
 378            _add_value(filepath, key, value, indent)
 379    # Check if it's a card
 380    elif key.upper() in pw_cards_upper:
 381        if key.upper() in present_keys_upper:
 382            _update_card(filepath, key, value, indent)
 383        else:
 384            _add_card(filepath, key, value, indent)
 385    else:
 386        raise ValueError(f'Unrecognised key: {key}')
 387    return None

 390def _update_value(
 391        filepath,
 392        key:str,
 393        value,
 394        indent:str='  '
 395        ) -> None:
 396    """Update the `value` of an existing `key` from a namelist. Called by `set_value()`.
 397
 398    Optionally change indentation with `indent`, 2 spaces by default.
 399    """
 400    key = key.strip()
 401    key_uncommented = key
 402    key_uncommented = key_uncommented.replace('(', r'\(')
 403    key_uncommented = key_uncommented.replace(')', r'\)')
 404    key_uncommented = rf'(?!\s*!\s*){key_uncommented}\s*='
 405    # Convert to int if necessary
 406    if key in _pw_int_values:
 407        value = int(value)
 408    line = f'{indent}{key} = {value}'
 409    if value == '':
 410        line = ''
 411    edit.replace_line(filepath=filepath, key=key_uncommented, text=line, replacements=1, regex=True)
 412    _update_other_values(filepath, key, value, indent)
 413    return None

 416def _add_value(
 417        filepath,
 418        key:str,
 419        value,
 420        indent:str='  '
 421    ) -> None:
 422    """Adds a `key` = `value` to a NAMELIST. Called by `set_value()`.
 423
 424    Optionally change indentation with `indent`, 2 spaces by default.
 425    """
 426    if value == '':  # No need to delete an unexisting value!
 427        return None
 428    if key in _pw_int_values:
 429        value = int(value)
 430    key = key.strip()
 431    value = str(value).strip()
 432    # Which namelist?
 433    parent_namelist = None
 434    for namelist in pw_namelists.keys():
 435        if key in pw_namelists[namelist]:
 436            parent_namelist = namelist
 437            break
 438    if not parent_namelist:
 439        raise ValueError(f"Key is not valid, '{key}' is not from any NAMELIST!")
 440    # Add the parent namelist if it does not exist, then add the value
 441    _add_namelist(filepath, parent_namelist)
 442    # Convert to int if necessary
 443    if value in _pw_int_values:
 444        value = int(value)
 445    line = f'{indent}{key} = {value}'
 446    parent_namelist_regex = rf'(?!\s*!\s*)({parent_namelist}|{parent_namelist.lower()})'
 447    edit.insert_under(filepath=filepath, key=parent_namelist_regex, text=line, insertions=1, regex=True)
 448    # Update other values if necessary
 449    _update_other_values(filepath, key, value, indent)
 450    return None

 453def _add_namelist(
 454        filepath,
 455        namelist:str,
 456    ) -> None:
 457    """Adds a `namelist` to the `filepath` if not present. Called by `add_value()` if needed."""
 458    namelists = list(pw_namelists.keys())
 459    namelist = namelist.upper().strip()
 460    if not namelist in namelists:
 461        raise ValueError(f'{namelist} is not a valid namelist! Valid namelists are:\n{namelists}')
 462    # Is the namelist already there?
 463    namelist_regex = rf'(?!\s*!\s*)({namelist}|{namelist.lower()})'
 464    is_present = find.lines(filepath=filepath, key=namelist_regex, regex=True)
 465    if is_present:
 466        return None
 467    # Find where to insert the namelist, from last to first.
 468    # We might have to insert it on top of the first CARD found.
 469    next_namelist =  _all_cards_regex
 470    namelists.reverse()
 471    # Get the present namelists, plus the target one
 472    present_namelists = []
 473    for section in namelists:
 474        is_section_present = find.lines(filepath=filepath, key=rf'(?!\s*!\s*)({section})', regex=True)
 475        if is_section_present or section.upper() == namelist.upper():
 476            present_namelists.append(section)
 477    # Get the very next section after the desired one
 478    for section in present_namelists:
 479        if section == namelist:
 480            break
 481        next_namelist = rf'(?!\s*!\s*)({section})(?!\s*=)'
 482    # Insert the target namelist on top of it!
 483    edit.insert_under(filepath, next_namelist, f'{namelist}\n/\n', 1, -1, True)
 484    return None

 487def _update_card(
 488        filepath,
 489        key:str,
 490        value:list,
 491        indent:str='  ',
 492    ) -> None:
 493    """Updates the `value` of a `key` CARD, present in `filepath`. Called by `set_value()`.
 494
 495    Optionally change indentation with `indent`, 2 spaces by default.
 496    """
 497    if value == '':
 498        return None
 499    value = normalize_card(value, indent)
 500    # Replace the CARD value up to the next CARD found
 501    key = key.upper().strip()
 502    key_uncommented = rf'(?!\s*!\s*)({key}|{key.lower()})'
 503    lines = '\n'.join(value)
 504    edit.replace_between(filepath=filepath, key1=key_uncommented, key2=_all_cards_regex, text=lines, delete_keys=False, regex=True)
 505    # We added the CARD below the previous CARD title, so we remove the previous CARD title
 506    edit.replace_line(filepath, key_uncommented, '', 1, 0, 0, True)
 507    # We might have to update other values, such as nat or ntyp
 508    _update_other_values(filepath, key, value, indent)
 509    return None

 512def _add_card(
 513        filepath,
 514        key:str,
 515        value:list,
 516        indent:str='  '
 517    ) -> None:
 518    """Adds a non-present `key` CARD with a given `value` to the `filepath`. Called by `set_value()`.
 519
 520    Optionally change indentation with `indent`, 2 spaces by default.
 521    """
 522    if value == '':
 523        return None
 524    value = normalize_card(value, indent)
 525    insert_value = value
 526    if isinstance(value, list):
 527        insert_value = '\n'.join(value)
 528    edit.insert_at(filepath, insert_value, -1)
 529    _update_other_values(filepath, key, value, indent)
 530    return None

 533def _update_other_values(
 534        filepath,
 535        key:str,
 536        value,
 537        indent:str='  '
 538    ) -> None:
 539    """Updates values that depend on the input value, eg. updates 'nat' when updating ATOMIC_POSITIONS.
 540
 541    Optionally change indentation with `indent`, 2 spaces by default.
 542    """
 543    file_path = file.get(filepath)
 544    old_values = read_in(file_path)
 545    # Key in upper cases for CARD values
 546    key = key.strip().upper()
 547    # CELL_PARAMETERS ?
 548    if key in ['CELL_PARAMETERS', 'CELL_PARAMETERS out']:
 549        if 'angstrom' in value[0] or 'bohr' in value[0]:
 550            edit.replace_line(file_path, r'(?!\s*!\s*)celldm\(\d\)\s*=', '', 0, 0, 0, True)
 551            edit.replace_line(file_path, r'(?!\s*!\s*)[ABC]\s*=', '', 0, 0, 0, True)
 552            edit.replace_line(file_path, r'(?!\s*!\s*)cos[AB][BC]\s*=', '', 0, 0, 0, True)
 553        elif 'alat' in value[0]:
 554            alat = extract.number(value[0])
 555            if alat:
 556                edit.replace_line(file_path, r'(?!\s*!\s*)CELL_PARAMETERS', 'CELL_PARAMETERS alat', -1, 0, 0, True)
 557                edit.replace_line(file_path, r'(?!\s*!\s*)celldm\(\d\)\s*=', f'celldm(1) = {alat}', 1, 0, 0, True)
 558        return None
 559    # ATOMIC_SPECIES ?
 560    elif key == 'ATOMIC_SPECIES':
 561        old_ntyp = old_values['ntyp']
 562        elements_found = []
 563        for line in value[1:]:
 564            element = extract.element(line)
 565            if element:
 566                if not element in elements_found:
 567                    elements_found.append(element)
 568        new_ntyp = len(elements_found)
 569        if old_ntyp != new_ntyp:
 570            set_value(filepath, 'ntyp', new_ntyp, indent)
 571        return None
 572    # ATOMIC_POSITIONS ?
 573    elif key in ['ATOMIC_POSITIONS', 'ATOMIC_POSITIONS out']:
 574        new_nat = len(value) - 1
 575        if old_values['nat'] != new_nat:
 576            set_value(filepath, 'nat', new_nat, indent)
 577        return None
 578    # Key in lower case for NAMELIST values
 579    key = key.lower()
 580    # Lattice params Angstroms?
 581    if key in ['a', 'b', 'c', 'cosab', 'cosac', 'cosbc']:
 582        edit.replace_line(file_path, r'(?!\s*!\s*)celldm\(\d\)\s*=', '', 0, 0, 0, True)
 583        edit.replace_line(file_path, r'(?!\s*!\s*)CELL_PARAMETERS', 'CELL_PARAMETERS alat', -1, 0, 0, True)
 584        return None
 585    # Lattice params Bohrs ?
 586    elif 'celldm' in key:
 587        edit.replace_line(file_path, r'(?!\s*!\s*)[ABC]\s*=', '', 0, 0, 0, True)
 588        edit.replace_line(file_path, r'(?!\s*!\s*)cos[AB][BC]\s*=', '', 0, 0, 0, True)
 589        edit.replace_line(file_path, r'(?!\s*!\s*)CELL_PARAMETERS', 'CELL_PARAMETERS alat', -1, 0, 0, True)
 590        return None
 591    return None

 594def add_atom(filepath, position, indent='  ') -> None:
 595    """Adds an atom in a given input `filepath` at a specified `position`.
 596
 597    Position must be a string or a list, as follows:
 598    `"specie:str float float float"` or `[specie:str, float, float, float]`.
 599
 600    This method updates automatically other related values,
 601    such as 'ntyp' when updating ATOMIC_SPECIES, etc.
 602
 603    Optionally change indentation with `indent`, 2 spaces by default.
 604    """
 605    new_atom = position
 606    if isinstance(position, list):
 607        if not len(position) == 4 or not isinstance(position[0], str):
 608            raise ValueError('If your atomic position is a list, it must contain the atom type and the three coords, as in [str, str/float, str/float, str/float]')
 609        new_atom = '   '.join(str(x) for x in position)
 610    elif not isinstance(position, str):
 611        raise ValueError(f'The specified position must be a list of size 4 (atom type and three coordinates) or an equivalent string! Your position was:\n{coords}')
 612    # Let's check that our new_atom makes sense
 613    atom = extract.element(new_atom)
 614    coords = extract.coords(new_atom)
 615    if not atom:
 616        raise ValueError(f'The specified position does not contain an atom at the beginning! Your position was:\n{position}')
 617    if len(coords) < 3:
 618        raise ValueError(f'Your position has len(coordinates) < 3, please check it.\nYour position was: {position}\nCoordinates detected: {coords}')
 619    if len(coords) > 3:
 620        coords = coords[:3]
 621    new_atom = f'{atom}   {coords[0]}   {coords[1]}   {coords[2]}'
 622    # Get the values from the file
 623    values = read_in(filepath)
 624    atomic_positions = values['ATOMIC_POSITIONS']
 625    atomic_positions.append(new_atom)
 626    # Update ATOMIC_POSITIONS. nat should be updated automatically.
 627    set_value(filepath=filepath, key='ATOMIC_POSITIONS', value=atomic_positions, indent=indent)
 628    # We might have to update ATOMIC_SPECIES!
 629    atomic_species = values['ATOMIC_SPECIES']
 630    is_atom_missing = True
 631    for specie in atomic_species[1:]:
 632        if atom == extract.element(specie):
 633            is_atom_missing = False
 634            break
 635    if is_atom_missing:  # Update ATOMIC_SPECIES. ntyp should be updated automatically.
 636        mass = periodictable.elements.symbol(atom).mass
 637        atomic_species.append(f'{atom}   {mass}   {atom}.upf')
 638        set_value(filepath=filepath, key='ATOMIC_SPECIES', value=atomic_species, indent=indent)
 639    return None

 642def get_atom(
 643        filepath:str,
 644        position:list,
 645        precision:int=3,
 646        return_anyway:bool=False,
 647    ) -> str:
 648    """Takes the approximate `position` of an atom, and returns the full line from the input `filepath`.
 649
 650    It compares the atomic positions rounded up to the specified `precision` decimals.
 651    If `return_anyway = True`, ignores errors and returns an empty string.
 652    """
 653    # Check that the coordinates are valid
 654    if isinstance(position, str):
 655        coordinates = extract.coords(position)
 656    elif isinstance(position, list):
 657        coordinates = position
 658        if len(position) == 1 and isinstance(position[0], str):  # In case someone like me introduces ['x, y, z']
 659            coordinates = extract.coords(position[0])
 660    else:
 661        if return_anyway:
 662            return ''
 663        raise ValueError(f'The atomic position must be a list or a string! Yours was:\n{position}\nDetected coordinates:\n{coordinates}')
 664    if len(coordinates) < 3:
 665        if return_anyway:
 666            return ''
 667        raise ValueError(f'Atomic position has less that 3 coordinates! Yours had len={len(coordinates)}:\n{position}\nDetected coordinates:\n{coordinates}')
 668    if len(coordinates) > 3:
 669        coordinates = coordinates[:3]
 670    # Round the input position up to the specified precision
 671    coordinates_rounded = []
 672    for coordinate in coordinates:
 673        coordinates_rounded.append(round(coordinate, precision))
 674    # Compare the rounded coordinates with the atomic positions
 675    content = read_in(filepath)
 676    if not 'ATOMIC_POSITIONS' in content.keys():
 677        if return_anyway:
 678            return ''
 679        raise ValueError(f'ATOMIC_POSITIONS missing in {filepath}')
 680    atomic_positions = content['ATOMIC_POSITIONS'][1:]  # Remove header
 681    matched_pos = None
 682    matched_index = None
 683    for i, atomic_position in enumerate(atomic_positions):
 684        coords =  extract.coords(atomic_position)
 685        coords_rounded = []
 686        for pos in coords:
 687            coords_rounded.append(round(pos, precision))
 688        if coordinates_rounded == coords_rounded:
 689            if matched_pos: # There was a previous match!
 690                if return_anyway:
 691                    return ''
 692                raise ValueError(f'More than one matching position found! Try again with more precision.\nSearched coordinates: {coordinates_rounded}')
 693            matched_pos = atomic_position
 694            matched_index = i
 695    if not matched_pos:
 696        if return_anyway:
 697            return ''
 698        raise ValueError(f'No matching position found! Try again with a less tight precision parameter.\nSearched coordinates: {coordinates_rounded}')
 699    # We must get the literal line, not the normalized one!
 700    atomic_positions_uncommented = rf'(?!\s*!\s*)(ATOMIC_POSITIONS|atomic_positions)'
 701    atomic_positions_lines = find.between(filepath=filepath, key1=atomic_positions_uncommented, key2=_all_cards_regex, include_keys=False, match=-1, regex=True)
 702    # Remove commented or empty lines
 703    atomic_positions_lines = atomic_positions_lines.splitlines()
 704    atomic_positions_cleaned = []
 705    for line in atomic_positions_lines:
 706        if line == '' or line.startswith('!') or line.startswith('#'):
 707            continue
 708        atomic_positions_cleaned.append(line)
 709    matched_line = atomic_positions_cleaned[matched_index]
 710    return matched_line.strip()

 713def normalize_card(card:list, indent='') -> list:
 714    """Take a matched card, and return it in a normalised format.
 715
 716    Optionally change indentation with `indent`, 0 spaces by default.
 717    """
 718    # Make sure it is a list!
 719    if isinstance(card, str):
 720        card = card.split('\n')
 721    if not isinstance(card, list):
 722        raise ValueError(f"Card must be a string or a list of strings! Yours was:\n{card}")
 723    # Keep the header
 724    cleaned_content = [card[0].strip()]
 725    for line in card[1:]:
 726        line = line.strip()
 727        if line == '' or line.startswith('!') or line.startswith('#'):
 728            continue
 729        elif any(key in line.upper() for key in pw_cards.keys()):
 730            break
 731        items = line.split()
 732        cleaned_line = f'{indent}{items[0].strip()}'  # Add the specified intent at the start of the line
 733        for item in items[1:]:
 734            item = item.strip()
 735            cleaned_line = cleaned_line + '   ' + item  # Three spaces between elements
 736        cleaned_content.append(cleaned_line)
 737    # Perform extra normalization for some CARDs
 738    if any(key in cleaned_content[0] for key in ['cell_parameters', 'CELL_PARAMETERS']):
 739        cleaned_content = _normalize_cell_parameters(cleaned_content, indent)
 740    elif any(key in cleaned_content[0] for key in ['atomic_positions', 'ATOMIC_POSITIONS']):
 741        cleaned_content = _normalize_atomic_positions(cleaned_content, indent)
 742    elif any(key in cleaned_content[0] for key in ['atomic_species', 'ATOMIC_SPECIES']):
 743        cleaned_content = _normalize_atomic_species(cleaned_content, indent)
 744    elif any(key in cleaned_content[0] for key in ['k_points', 'K_POINTS']):
 745        cleaned_content = _normalize_k_points(cleaned_content)
 746    return cleaned_content

 749def _normalize_cell_parameters(card, indent='') -> list:
 750    """Performs extra formatting to a previously cleaned CELL_PARAMETERS `card`.
 751
 752    Optionally change indentation with `indent`, 0 spaces by default.
 753    """

 755        return None

 771        alat = extract.number(cell_parameters[0], 'alat')

 775            cell_parameters[0] = 'CELL_PARAMETERS alat'
 776    else:
 777        raise ValueError(f'CELL_PARAMETERS must contain angstrom, bohr or alat! Yours was:\n{cell_parameters}')
 778    return cell_parameters

 900def resume(
 901        folder=None,
 902        in_str:str='.in',
 903        out_str:str='.out',
 904    ) -> None:
 905    """Update an input file with the atomic coordinates of an output file.
 906
 907    This can be used to quickly resume an unfinished or interrupted calculation.
 908
 909    Takes a `folder` from a QE calculation, CWD if empty.
 910    Input and output files are determined automatically,
 911    but must be specified with `in_str` and `out_str`
 912    if more than one file ends with `.in` or `.out`.
 913
 914    Old input and output files will be renamed automatically.
 915    """
 916    folder = file.get_dir(folder)
 917    exclude = ['resumed', 'slurm-']
 918    input_file = file.get(folder, include=in_str, exclude=exclude)
 919    output_file = file.get(folder, out_str, exclude=exclude)
 920    if not input_file or not output_file:
 921        raise FileNotFoundError('Missing input or output file')
 922    dict_out = read_out(output_file)
 923    atomic_positions = dict_out.get('ATOMIC_POSITIONS out')
 924    cell_parameters = dict_out.get('CELL_PARAMETERS out')
 925    if not atomic_positions:
 926        raise ValueError(f'Missing atomic positions in output file {output_file}')
 927    # Backup old files
 928    file.backup(input_file, keep=True, label='resumed')
 929    file.backup(output_file, keep=False, label='resumed')
 930    # Update input file
 931    set_value(input_file, 'ATOMIC_POSITIONS', atomic_positions)
 932    if cell_parameters:
 933        set_value(input_file, 'CELL_PARAMETERS', cell_parameters)
 934    return None

 980def to_cartesian(filepath, coordinates) -> str:
 981    """Converts a given `cordinates` from crystal lattice vectors to cartesian.
 982

 990    converted_coords = np.matmul(cell_base, coords)

 992    final_coords = converted_coords_row.tolist()
 993    #print(f'FINAL_COORDINATES: {final_coords}')
 994    return final_coords

 997def from_cartesian(filepath, coordinates:list) -> str:
 998    """Converts a given `cordinates` from cartesian to the base of lattice vectors.
 999
1000    Only for `ibrav=0`. Uses the cell parameters from the input `filepath`.

1010    final_coords = converted_coords_row.tolist()
1011    #print(f'FINAL_COORDINATES: {final_coords}')  # DEBUG
1012    return final_coords
